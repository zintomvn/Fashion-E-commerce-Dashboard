\subsection{Dashboard A — Tổng quan lợi nhuận \& dự báo (Revenue/COGS/Gross Profit + RFM + BCG)}

\subsubsection{Cơ sở lý thuyết}
\begin{itemize}
    \item \textbf{Chỉ số tài chính cơ bản.} Doanh thu (Revenue), giá vốn (COGS), lợi nhuận gộp (Gross Profit) và biên lợi nhuận gộp (Gross Margin) là các chỉ số nền tảng để đánh giá hiệu quả kinh doanh.
    \begin{equation}
        \text{Gross Profit} = \text{Revenue} - \text{COGS},\quad
        \text{Gross Margin} = \frac{\text{Gross Profit}}{\text{Revenue}}.
    \end{equation}

    \item \textbf{Phân tích chuỗi thời gian và dự báo.} Dữ liệu doanh thu/chi phí thường có \emph{xu hướng} và \emph{mùa vụ}. Các mô hình cây tăng cường (Gradient Boosting như XGBoost/LightGBM) tận dụng các \emph{đặc trưng lịch} (tháng, thứ, ngày trong năm, kỳ nghỉ) để dự báo tốt hơn so với mô hình tuyến tính khi quan hệ là phi tuyến.

    \item \textbf{RFM Segmentation.} RFM (Recency--Frequency--Monetary) là phương pháp phân khúc khách hàng dựa trên: độ gần lần mua gần nhất, tần suất mua, và giá trị chi tiêu. Có thể dùng phân cụm (ví dụ K-means) để nhóm khách hàng theo hành vi.

    \item \textbf{Ma trận BCG/Portfolio.} Phân loại danh mục sản phẩm thành \emph{Stars, Cash Cows, Question Marks, Dogs} dựa trên trục giá trị (profit/margin) và trục quy mô (volume), phục vụ quyết định đầu tư/tối ưu danh mục.
\end{itemize}

\paragraph{(1) Xu hướng \& dự báo doanh thu/giá vốn/lợi nhuận gộp}
\textbf{Giả thuyết (Hypothesis).} Revenue và COGS có mùa vụ theo lịch (tháng, thứ trong tuần) và chịu tác động bởi các dịp cao điểm; do đó mô hình boosting với đặc trưng lịch có thể dự báo tốt hơn.

\textbf{Feature engineering.} Tạo các biến lịch (tháng, thứ, ngày trong năm), các biến chu kỳ (sin/cos) và các biến tương tác lịch (nếu có).

\textbf{Huấn luyện (Train).} Dùng mô hình boosting (ví dụ XGBoost) và chia tập theo thời gian (time-series split) để tránh rò rỉ tương lai.

\textbf{Tuning.} Tối ưu siêu tham số để chọn mô hình có sai số thấp nhất; kết quả dự báo được xuất ra \texttt{outputs/submission\_xgb\_fold5\_tuned.csv}.

\textbf{Giải thích (Explain).} Ưu tiên giải thích bằng mức độ quan trọng đặc trưng (feature importance) và kiểm tra tính hợp lý theo mùa vụ trên Hình~\ref{fig:dbA_rev_cogs_forecast}.

\textbf{Insight.} Kết hợp Hình~\ref{fig:dbA_rev_cogs_forecast} và Hình~\ref{fig:dbA_gross_profit} giúp phát hiện giai đoạn \emph{doanh thu tăng nhưng biên lợi nhuận giảm} (COGS tăng nhanh hơn), từ đó ưu tiên kiểm soát giá vốn/chiết khấu trước khi tăng trưởng doanh thu.

\textbf{Phương pháp.} Dùng bộ ba Recency--Frequency--Monetary, sau đó phân cụm để phân loại các nhóm như \emph{Champions, Loyal Customers, Bargain Hunters, At Risk/Churn, Hibernating/Low Engagement}.

\textbf{Nhận xét.} Hình~\ref{fig:dbA_rfm_cluster} cho thấy sự phân tách giữa nhóm giá trị cao (tần suất/chi tiêu cao, recency thấp) và nhóm có nguy cơ rời bỏ.

\textbf{Insight.} Các nhóm \emph{Champions/Loyal Customers} nên là mục tiêu của chương trình giữ chân (membership, ưu đãi riêng) trong khi nhóm \emph{At Risk/Churn} cần can thiệp tái kích hoạt (re-activation) với điều kiện khuyến mãi/ưu đãi được \emph{cá nhân hoá} (liên hệ Dashboard B).

\textbf{Nhận xét.} Hình~\ref{fig:dbA_bcg} xác định nhóm sản phẩm cần đầu tư (Stars) và nhóm cần tối ưu vòng đời (Cash Cows). Hình~\ref{fig:dbA_segment_matrix} chỉ ra mối liên hệ giữa nhóm khách hàng và cấu trúc danh mục.

\textbf{Insight.} Nếu nhóm khách hàng giá trị cao (Champions/Loyal) tập trung mua \emph{Cash Cows/Stars} (Hình~\ref{fig:dbA_segment_matrix}), doanh nghiệp nên ưu tiên tồn kho và trải nghiệm giao hàng cho các SKU thuộc hai nhóm này để bảo vệ lợi nhuận.

\subsection{Dashboard B — Khuyến mãi: Uplift targeting \& mô phỏng lợi nhuận tăng thêm}

\subsubsection{Cơ sở lý thuyết}
\begin{itemize}
    \item \textbf{Vấn đề hiệu ứng dị biệt (Heterogeneous Treatment Effect).} Khuyến mãi không tác động đồng đều: cùng một ưu đãi có thể tăng xác suất mua ở nhóm này nhưng không hiệu quả (thậm chí phản tác dụng) ở nhóm khác.

    \item \textbf{Uplift modeling.} Mục tiêu là ước lượng \emph{tác động nhân quả cá nhân hoá}:
    \begin{equation}
        \text{Uplift}(x) = P(Y{=}1\mid T{=}1, x) - P(Y{=}1\mid T{=}0, x).
    \end{equation}
    Trong đó $T$ là treatment (có khuyến mãi/không), $Y$ là kết quả (mua/không mua), $x$ là đặc trưng.

    \item \textbf{T-Learner.} Huấn luyện hai mô hình tách biệt cho $T{=}1$ và $T{=}0$, sau đó lấy chênh lệch xác suất làm uplift.

    \item \textbf{Mô phỏng P\&L kỳ vọng (expected profit).} Giữ nguyên xác suất mua từ mô hình, thay đổi lợi nhuận biên theo mức giảm giá:
    \begin{align}
        m_{base} &= price - cogs,\\
        m_{promo} &= promo\_price - cogs,\\
        E[\pi_{base}] &= P(Y\mid NoPromo)\cdot m_{base},\\
        E[\pi_{promo}] &= P(Y\mid Promo)\cdot m_{promo},\\
        \Delta E[\pi] &= E[\pi_{promo}] - E[\pi_{base}].
    \end{align}
    Có thể điều chỉnh phạt theo \emph{return rate} nếu muốn giảm rủi ro hoàn hàng.
\end{itemize}

\subsubsection{Phân tích chi tiết}
\paragraph{(1) Uplift distribution \& chọn nhóm mục tiêu}
\textbf{Giả thuyết (Hypothesis).} Tác động khuyến mãi phụ thuộc phân khúc khách hàng và danh mục; vì vậy phân phối uplift theo nhóm sẽ có độ phân tán lớn (heterogeneity).

\textbf{Feature engineering.} Sử dụng đặc trưng hành vi và ngữ cảnh mua hàng: RFM, lịch sử nhận khuyến mãi, thuộc tính sản phẩm (category/segment/size/color/price/cogs), và các biến kênh (device, acquisition channel, payment\ldots).

\textbf{Huấn luyện (Train).} T-Learner với hai mô hình LightGBM: một cho nhóm $T{=}1$ (có promo) và một cho $T{=}0$ (không promo). Dùng \texttt{scale\_pos\_weight} để xử lý mất cân bằng lớp (imbalanced).

\textbf{Tuning.} Tối ưu siêu tham số để tăng AUC/giảm log-loss (classification) và cải thiện tổng thể chất lượng dự báo $P(Y|T,x)$.

\textbf{Giải thích (Explain).} Dùng Dashboard D (SHAP, correlation, confounders) để giải thích \emph{tại sao} uplift cao/thấp theo đặc trưng.

\textbf{Nhận xét.} Hình~\ref{fig:dbB_uplift_violin} cho thấy uplift có thể âm ở một số nhóm (khuyến mãi làm giảm xác suất mua hoặc làm giảm lợi nhuận kỳ vọng), do đó không nên triển khai đại trà.

\textbf{Insight.} Ưu tiên nhắm nhóm/segment có phân phối uplift dương rõ rệt (Hình~\ref{fig:dbB_uplift_violin}) và có đóng góp lợi nhuận tăng thêm cao (Hình~\ref{fig:dbB_profit_by_group}), thay vì tối đa số lượng khách hàng được giảm giá.

\paragraph{(2) Gain curve \& tối ưu ngưỡng nhắm mục tiêu}
\textbf{Phương pháp.} Sắp xếp khách hàng theo \texttt{expected\_uplift\_profit} hoặc \texttt{Uplift\_Score} và vẽ lợi nhuận tăng thêm lũy kế (gain curve).

\textbf{Nhận xét.} Hình~\ref{fig:dbB_gain_curve} thể hiện hiệu quả biên (marginal gain) khi mở rộng tập mục tiêu. Khi đường cong bắt đầu “phẳng”, thêm khách hàng mới không còn đem lại lợi nhuận tăng thêm đáng kể.

\textbf{Insight.} Chọn ngưỡng uplift hoặc Top-N tại điểm gãy của đường cong (Hình~\ref{fig:dbB_gain_curve}) giúp cân bằng giữa quy mô chiến dịch và ROI, tránh giảm giá cho nhóm có uplift thấp.

\paragraph{(3) Waterfall P\&L \& trade-off mức giảm giá}
\textbf{Phương pháp.} Tách hai hiệu ứng: (i) \emph{margin effect} do giảm giá làm giảm biên lợi nhuận; (ii) \emph{uplift effect} do tăng xác suất mua. Tổng hợp trong waterfall.

\textbf{Nhận xét.} Hình~\ref{fig:dbB_promo_waterfall} giúp kiểm tra trường hợp ``uplift dương nhưng net âm'' (giảm giá quá mạnh khiến margin effect lớn hơn uplift effect).

\textbf{Insight.} Mức giảm giá tối ưu thường không phải lớn nhất; cần chọn mức giảm sao cho \emph{uplift effect} vượt \emph{margin effect} (Hình~\ref{fig:dbB_promo_waterfall}) và ưu tiên nhắm nhóm có return rate thấp để giảm rủi ro (phạt theo return\_rate).

\textbf{Bảng sử dụng (hiển thị):} \texttt{outputs/uplift\_predictions\_test\_set.csv} (các cột \texttt{True\_T}, \texttt{True\_Y}, \texttt{P\_Y\_given\_Promo}, \texttt{P\_Y\_given\_NoPromo}, \texttt{Uplift\_Score}).\\

\textbf{Cơ sở lý thuyết bổ sung.}
\begin{itemize}
    \item \textbf{Surrogate model.} Thay vì diễn giải trực tiếp mô hình gốc (có thể phức tạp), huấn luyện mô hình thay thế để xấp xỉ đầu ra (ví dụ Uplift\_Score) và dùng nó để giải thích.
    \item \textbf{SHAP (Shapley Additive Explanations).} Đánh giá đóng góp trung bình của từng đặc trưng tới dự đoán; dùng mean(|SHAP|) để xếp hạng biến quan trọng.
    \item \textbf{Biến gây nhiễu (Confounders).} Một biến $f$ là gây nhiễu nếu tương quan đồng thời với treatment $T$ và outcome $Y$. Điểm gây nhiễu được tính:
    \begin{equation}
        score(f)=|corr(f,T)|\times |corr(f,Y)|.
    \end{equation}
\end{itemize}

\subsubsection{Phân tích chi tiết}
\paragraph{(1) Xếp hạng hành động theo \% thay đổi lợi nhuận}
\textbf{Giả thuyết (Hypothesis).} Các phân khúc khác nhau có ``nút thắt'' khác nhau: nhóm giá trị cao nhạy với tối ưu COGS/price, trong khi nhóm trung thành có thể nhạy với tình trạng hết hàng (stockout).

\textbf{Feature engineering.} Chuẩn hoá dữ liệu theo phân khúc; xây dựng các biến phản ánh vận hành: \texttt{cogs}, \texttt{price}, \texttt{stockout\_flag}, \texttt{stockout\_days}, \texttt{shipping\_fee\_allocated}, \texttt{return\_quantity}, \ldots

\textbf{Huấn luyện (Train).} Huấn luyện các mô hình thành phần để dự báo \texttt{unit\_price}, \texttt{quantity}, \texttt{cost}. Sau đó tạo kịch bản phảnfactual bằng cách thay đổi một biến trong ``vùng p95'' của phân khúc.

\textbf{Tuning.} Chọn mô hình thắng theo $R^2$ (ví dụ Random Forest cho \texttt{unit\_price}/\texttt{cost} ở nhiều phân khúc) và tinh chỉnh để giảm RMSE.

\textbf{Giải thích (Explain).} Đọc bảng MI/feature importance (Hình~\ref{fig:dbC_mi}, Hình~\ref{fig:dbC_fi_compare}) để xác định biến tác động chính.

\textbf{Nhận xét.} Biểu đồ Hình~\ref{fig:dbC_delta_profit} cho thấy các hành động có thể tạo ra mức thay đổi lợi nhuận rất khác nhau tuỳ phân khúc.

\textbf{Insight.} Dựa trên bảng hành động tốt nhất (Hình~\ref{fig:dbC_leaderboard}), có thể rút ra quy tắc vận hành theo phân khúc (ví dụ: giảm \texttt{cogs} là đòn bẩy mạnh cho nhiều phân khúc; riêng nhóm \emph{Loyal Customers} cải thiện bằng cách giảm stockout).

\paragraph{(2) Waterfall: lợi nhuận tăng do doanh thu hay do chi phí?}
\textbf{Phương pháp.} Mỗi kịch bản có các thành phần: \texttt{baseline\_revenue}, \texttt{scenario\_revenue}, \texttt{baseline\_cost}, \texttt{scenario\_cost}. Waterfall (Hình~\ref{fig:dbC_waterfall}) giúp phân rã:
\begin{equation}
    \Delta Profit = \Delta Revenue - \Delta Cost.
\end{equation}

\textbf{Nhận xét.} Nếu một kịch bản tăng lợi nhuận chủ yếu nhờ \emph{giảm chi phí} (\(\Delta Cost<0\)), doanh nghiệp cần kiểm tra tính khả thi (đàm phán nhà cung cấp, tối ưu vận chuyển\ldots).

\textbf{Insight.} Waterfall (Hình~\ref{fig:dbC_waterfall}) giúp ưu tiên các kịch bản ``an toàn'' (tăng lợi nhuận mà không phụ thuộc vào giả định tăng doanh thu khó đạt).

\paragraph{(3) Chất lượng mô hình thành phần \& hàm ý tin cậy}
\textbf{Nhận xét.} Từ metrics (Hình~\ref{fig:dbC_component_metrics}), các mục tiêu như \texttt{unit\_price} và \texttt{cost} thường có $R^2$ cao, trong khi \texttt{quantity} có thể khó dự báo (R² thấp/âm), do nhu cầu biến động mạnh và nhiễu.

\textbf{Insight.} Các quyết định tối ưu nên ưu tiên đòn bẩy dựa trên biến có thể mô hình hoá ổn định (ví dụ tối ưu \texttt{cogs}/\texttt{price}), và thận trọng với kịch bản phụ thuộc nhiều vào dự báo \texttt{quantity} khi chất lượng mô hình thấp.
